* 5.1.3 A Grammar for Lox expressions

#+begin_src text
  expression	→ literal
                  | unary
                  | binary
                  | grouping ;

  literal         → NUMBER | STRING | "true" | "false" | "nil" ;
  grouping        → "(" expression ")" ;
  unary           → ( "-" | "!" ) expression ;
  binary          → expression operator expression ;
  operator        → "==" | "!=" | "<" | "<=" | ">" | ">="
                  | "+" | "-" | "*" | "/" ;
#+end_src
